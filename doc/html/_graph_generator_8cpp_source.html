119 set<tuple<int,int,int>> GraphGenerator::get_edges_graph(Graph *g)
120 {
121  set<tuple<int,int,int>> edges;
122  for (int i = 0; i < igraph_ecount(g); i++) {
123  int from = IGRAPH_FROM(g, i);
124  int to = IGRAPH_TO(g, i);
125  if (igraph_cattribute_has_attr(g,IGRAPH_ATTRIBUTE_EDGE,"weight"))
126  {
127  /* edge attributes */
128  igraph_vector_t gtypes, vtypes, etypes;
129  igraph_strvector_t gnames, vnames, enames;
130 
131  igraph_vector_init(&gtypes,0);
132  igraph_vector_init(&vtypes,0);
133  igraph_vector_init(&etypes,0);
134  igraph_strvector_init(&gnames,0);
135  igraph_strvector_init(&vnames,0);
136  igraph_strvector_init(&enames,0);
137 
138  igraph_cattribute_list(g,&gnames,&gtypes,&vnames,&vtypes,&enames, &etypes);
139  int cost;
140  for (int j = 0; j < igraph_strvector_size(&enames); j++) {
141  cost = EAN(g, STR(enames, j), i);
142  }
143  edges.insert(make_tuple(from,to,cost));
144  }else
145  {
146  edges.insert(make_tuple(from,to,0));
147  }
148  }
149  return edges;
150 }

156 set<int> GraphGenerator::get_neighbors(Graph *g, int id)
157 {
158  set<int> neighbors;
159  VertexIter vit;
160  VertexSelect vs;
161  igraph_vs_adj(&vs, id, IGRAPH_ALL);
162  igraph_vit_create(g, vs, &vit);
163  while (!IGRAPH_VIT_END(vit)) {
164  int v_id = IGRAPH_VIT_GET(vit);
165  neighbors.insert(v_id);
166  IGRAPH_VIT_NEXT(vit);
167  }
168  igraph_vit_destroy(&vit);
169  igraph_vs_destroy(&vs);
170  return neighbors;
171 }
