35 map<string, string> parse_parameters(string params_file)
36 {
37  ifstream file(params_file);
38  if ( !file.is_open() )
39  {
40  cerr << "error : failed to open parameters file !" << endl;
41  exit(EXIT_FAILURE);
42  }
43  map<string, string> parameters;
44  string line;
45  while ( getline(file, line) )
46  {
47  if ( !line.find("//") || line.empty() )
48  continue;
49  istringstream iss(line);
50  string name, value;
51  iss >> name;
52  iss >> value;
53  parameters[name] = value;
54  if ( file.bad() )
55  {
56  cerr << "error : while reading the parameters file !" << endl;
57  exit(EXIT_FAILURE);
58  }
59  }
60  file.close();
61  return parameters;
62 }
